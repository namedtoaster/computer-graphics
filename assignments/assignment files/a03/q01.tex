(a)

\vv{3}{1/2} + \vv{-5/2}{3/4}

&= \vv{15 -(-35)}{15/2 - 28}\\

ANSWER: \boxed{\vv{50}{-41/2}}

(d)

\end{align*}
ANSWER: \boxed{\vv{-25/2}{141/4}}

(e) 
\begin{align*}

&= -21 + 21/4\\
&= -63/4
\end{align*}
ANSWER: \boxed{-63/4}

(f) We have
\begin{align*}
\vv{3}{1/2} \cdot \vv{2}{3} &= \| \vv{3}{1/2} \|  
                               \cdot 
                               \| \vv{2}{3} \| 
                               \cdot
                               \cos \theta
\\
\THEREFORE \cos\theta     &= \frac {\vv{3}{1/2} \cdot \vv{2}{3}} 
                               {\| \vv{3}{1/2} \|  \cdot \| \vv{2}{3} \| }

                          &= \frac{15/2}{\sqrt{481/4}}
\end{align*}


ANSWER: \boxed{\frac{15/2}{\sqrt{481/4}}}

(h)

\begin{align*}
